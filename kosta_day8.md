//포인터 배열 위치 바꿔주는 것
void SWAP(void* a, void* b, int op)
{
	if (op == 1)
	{
		char c;
		c = *(char*)a;
		*(char*)a = *(char*)b;
		*(char*)b = c;
	}
	else if (op == 4)
	{
		int c;
		c = *(int*)a;
		*(int*)a = *(int*)b;
		*(int*)b = c;
	}
	else if (op == 8)
	{
		double c;
		c = *(double*)a;
		*(double*)a = *(double*)b;
		*(double*)b = c;
	}



}
